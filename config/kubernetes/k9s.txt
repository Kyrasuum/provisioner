  # Hitting Shift-1 navigates to your deployments
  shift-1:

  # Hitting Shift-2 navigates to your secrets
  shift-2:
    shortCut:    Shift-2
    description: View secrets
    command:     secrets

  # Hitting q will exit current view
  q:
    shortCut:    q
    description: exit
    command:     quit
